function handleCommunication() { conn.on('open', () => { document.getElementById('status').innerText = "🟢 Connected to Peer!"; }); // When a file is received conn.on('data', (data) => { const blob = new Blob([data.file], { type: data.type }); const url = URL.createObjectURL(blob); document.getElementById('download-area').style.display = "block"; document.getElementById('file-info').innerText = "Received: " + data.name; const dlBtn = document.getElementById('download-btn'); dlBtn.href = url; dlBtn.download = data.name; document.getElementById('status').innerText = "📩 File Received!"; }); conn.on('close', () => { document.getElementById('status').innerText = "❌ Disconnected"; }); } function sendFile() { const fileInput = document.getElementById('file-input'); const file = fileInput.files[0]; if (!conn || !conn.open) return alert("Connect to a friend first!"); if (!file) return alert("Please select a file to send"); document.getElementById('status').innerText = "📤 Sending: " + file.name; conn.send({ file: file, name: file.name, type: file.type }); setTimeout(() => { document.getElementById('status').innerText = "✅ Sent Successfully!"; }, 2000); } // Handle network errors automatically peer.on('error', (err) => { document.getElementById('status').innerText = "⚠️ Connection Error"; console.error('PeerJS Error:', err); });
